    问题答案接口

    接口类型 Protobuf

    请求地址 /im-qa/answer

    请求方式 POST

    请求Body

    //自动问答请求
message GetQARequest {
    required int32 qaId = 1;//问答id
}

    响应格式 Protobuf

    响应Body

    //自动问答返回
message GetQAResponse {
    required int32 type = 1;//消息类型
    optional bytes  msg = 2;//消息体
}
