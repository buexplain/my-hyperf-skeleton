汇聊接口文档

    问题列表接口

    接口类型 Protobuf

    请求地址 /im-qa/question

    请求方式 POST

    请求Body

    //请求问题列表
message QAListRequest {
    required int32 client = 1;//平台类型 1:PC, 2:APP
}

    响应格式 Protobuf

    响应Body

    //自动问答问题列表返回
message QAListResponse {
    repeated QAItem qaList = 1;
}

//自动问答问题
message QAItem
{
    required int32 id= 1;//问题编号
    required string question = 2;//问题描述
}
