    syntax = "proto2";

package App.Protobuf.QuestionAnswer;

enum IMMessageType {
    AnalysisStock = 1; //诊股消息
    AutoReply = 2; //自动问答消息
    RESERVE3 = 3;
    RESERVE4 = 4;
    RESERVE5 = 5;
    RESERVE6 = 6;
    RESERVE7 = 7;
    RESERVE8 = 8;
    RESERVE9 = 9;
    RESERVE10 = 10;
    Text = 11;
    Voice = 12;
    Image = 13;
    Video = 14;
    File = 15;
    Question = 16;
}

//请求问题列表
message QAListRequest {
    required int32 client = 1;//平台类型 1:PC, 2:APP
}
//自动问答问题列表返回
message QAListResponse {
    repeated QAItem qaList = 1;
}

//自动问答问题
message QAItem
{
    required int32 id= 1;//问题编号
    required string question = 2;//问题描述
}

//自动问答请求
message GetQARequest {
    required int32 qaId = 1;//问答id
}

//自动问答返回
message GetQAResponse {
    required int32 type = 1;//消息类型
    optional bytes  msg = 2;//消息体
}

//文本消息
message TextMessage {
    required string content = 1;
}

//语音消息
message VoiceMessage {
    required string displayName = 1;
    required uint32 duration = 2;
    optional string localUrl = 3;
    optional string remoteUrl = 4;
}

message ImageMessage {
    required string displayName = 1;
    optional uint32 width = 2;
    optional uint32 height = 3;
    optional string localUrl = 4;
    optional string remoteUrl = 5;
}

message VideoMessage {
    required string displayName = 1;
    required fixed64 fileLength = 2;
    required uint32 duration = 3;
    optional uint32 width = 4;
    optional uint32 heigh = 5;
    optional string localUrl = 6;
    optional string remoteUrl = 7;
}

message FileMessage {
    required string displayName = 1;
    optional string localUrl = 2;
    optional string remoteUrl = 3;
    optional fixed64 fileLength = 4;
}

//诊股消息
message DiagnosticMessage{
    required string  stockName = 1;
    required string  stockCode = 2;
    required float   costPrice = 3;     //成本价
    required string  positon = 4;       //仓位
}
